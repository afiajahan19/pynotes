# Fix the income bracket
def label_fix(label):
    label = label.strip().lower()
    if(label == '<=50k'):
        return 0
    return 1

census['income'] = census['income'].apply(label_fix)

census.rename(columns={"income": "income_bracket"}, inplace=True)
