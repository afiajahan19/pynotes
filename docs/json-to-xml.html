json_data = {
    "person": {
        "name": "Alice",
        "age": 30,
        "phones": ["123-4567", "987-6543"]
    }
}

import xmltodict

# Convert JSON (Python dict) to XML
xml_output = xmltodict.unparse({"root": json_data}, pretty=True)
print(xml_output)
