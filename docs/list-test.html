users = [
    ('Anders', 'A', 21),
    ('Balmer', 'B', 45),
    ('Stephe', 'S', 24)
]

print(users)

[('Anders', 'A', 21), ('Balmer', 'B', 45), ('Stephe', 'S', 24)]

sorted(users, key = lambda x: x[2])

[('Anders', 'A', 21), ('Stephe', 'S', 24), ('Balmer', 'B', 45)]

sorted(users, key = lambda x: -x[2])

[('Balmer', 'B', 45), ('Stephe', 'S', 24), ('Anders', 'A', 21)]
